#first change: working directory through R Studio options #variable fileToRead with name of filefileToRead<-"~/Documents/COURSERA/4.ExploratoryDataAnalysis/project1/household_power_consumption.txt"#reading downloaded dataset in data variabledata <-read.table(fileToRead, header=TRUE, sep=";",stringsAsFactors=FALSE, dec=".")#checking previous size estimation: nr rows*nr columns* 8 bytes/numericobject.size(data)head(data)#Just dates of interest in info variableinfo <- data[data$Date %in% c("1/2/2007", "2/2/2007"),]object.size(info)#variables to plotdateTime <- strptime(paste(info$Date, info$Time, sep=" "), "%d/%m/%Y %H:%M:%S") gAp <- as.numeric(info$Global_active_power)#plotting in png file Global Active Power png("plot2.png", width = 480, height= 480)plot(dateTime, gAp, type="l", xlab="", ylab="Global Active Power (kilowatts)")dev.off()#preview on screenplot(dateTime, gAp, type="l", xlab="", ylab="Global Active Power (kilowatts)")
